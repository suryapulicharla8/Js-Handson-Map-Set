        Your task is to remove all the duplicate occurrences of characters in the string.
        
        Input:
        abcadeecfb
        
        Output:
        abcdef"
    
 ANS.
        let s="abcdefabcef";
        let set=new Set();
        for(let i=0;i < s.length;i++){
            set.add(s[i])
        }
        let res=" ";
        for(let ch of set){
            res=res+ch;
        }
        console.log(res);
